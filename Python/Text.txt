d = int(input())
total = 0
s = set()
while total != d:
    a = str(input().strip().lower())
    s.add(a)
    total+=1
l = int(input())
total = 0
ln = []
while total != l:
    a = input().strip().lower().split()
    ln += a
    total+=1
s1 = set()
for i in ln:
    if i not in s1:
        s1.add(i)
print(*(s1-s), sep = '\n')
