a = str(input())
b = str(input())
s = str(input())
ss = str(input())
s1, s2 = '', ''
d = dict(zip(a,b))
for i in s:
    for key in d.keys():
        if i == key:
            s1 += d.get(key)
            break
for i in ss:
    for key, value in d.items():
        if i == value:
            s2 += key
            break
print(s1, '\n' + s2)
